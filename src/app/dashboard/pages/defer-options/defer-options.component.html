on interaction - Cuando interactuas con el propio elemento dentro del defer

@defer (on interaction) { on interaction }@placeholder {
Click en el div
}

on interaction(element) - Cuando interactuas con otro elemento

@defer (on interaction(btnInteraction)) { on interaction (element) }@placeholder {
Placeholder Div
}

on hover - Cuando pasas por encima del propio elemento dentro del defer

@defer (on hover) { on hover }@placeholder {
Placeholder Div
}

on hover(element) - Cuando pasas por encima de otro elemento

@defer (on hover(btnHover)) { on hover (element) }@placeholder {
Placeholder Div
}

on immediate - En cuanto se pueda se carga

@defer (on immediate) { on immediate }@placeholder {
Placeholder Div
}

on timer (time en milis) - Se carga cuando pasa el tiempo estipulado

@defer (on timer(2000)) { on timer(2000) }@placeholder {
Placeholder Div
}

on hover; prefetch on idle - Se carga cuando pasas por encima del elemento, pero lo precarga en idle

@defer (on hover; prefetch on idle) { on hover; prefetch on idle }@placeholder {
Placeholder Div
}
